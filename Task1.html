import sklearn

def createPipeLine(X, model):
  # Preprocessing for numerical data
  numerical_transformer = SimpleImputer(strategy='median')

  # Preprocessing for categorical data
  categorical_transformer = Pipeline(steps=[
      ('imputer', SimpleImputer(strategy='most_frequent')),
      ('onehot', OneHotEncoder(handle_unknown='ignore'))
  ])


  numerical_cols = [cname for cname in X.columns if X[cname].dtype in ['int64', 'float64']]
  categorical_cols = [cname for cname in X.columns if X[cname].nunique() < 10 and 
                      X[cname].dtype == "object"]


  # Bundle preprocessing for numerical and categorical data
  preprocessor = ColumnTransformer(
      transformers=[
          ('num', numerical_transformer, numerical_cols),
          ('cat', categorical_transformer, categorical_cols)
      ])

  #model = linear_model.SGDClassifier(loss='log', alpha=0, learning_rate='constant', eta0=0.01)


  my_pipeline = Pipeline(steps=[('preprocessor', preprocessor),
                                ('Scaler', StandardScaler()),
                                ('model', model)
                              ])
  return my_pipeline

from sklearn.preprocessing import OneHotEncoder

encoder = None


def doOHE(df, cols):
  global encoder
  if encoder is None:
    OH_encoder = OneHotEncoder(handle_unknown='ignore', sparse=False)
    OH_encoder = OH_encoder.fit(df[cols])
    encoder = OH_encoder  
  OH_cols = pd.DataFrame(encoder.transform(df[cols]))
  OH_cols.columns = encoder.get_feature_names(cols)
  OH_cols.index = df.index

  df = df.drop(cols,axis=1)
  df = pd.concat([df, OH_cols],axis=1)
  return df

def addFeatures(df):
  df["Underage"] = (df['Age'] < 18).map({False:0.0,True:1.0})
  df["TotalFamily"] = df['SibSp'] + df['Parch']
  df['IsAlone'] = (df['TotalFamily'] == 0).map({False:0.0,True:1.0})
  df['Age_Squared'] = df['Age']**2
  return df
